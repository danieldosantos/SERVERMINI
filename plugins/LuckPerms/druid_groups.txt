# LuckPerms configuração de grupos para classe Druida
lp creategroup class_druid
lp group class_druid setdisplayname "Druida"
lp group class_druid meta add prefix "&2[Druida]&f "
lp group class_druid permission set class_druid true
lp group class_druid permission set skillapi.class.druid true
lp group class_druid permission set mythicmobs.item.druid true
lp group class_druid permission set skript.command.forma true
lp group class_druid permission set skript.command.kit_druida true
lp group class_druid permission set placeholderapi.placeholder.skills_wildshape_time true
lp group class_druid permission set placeholderapi.placeholder.skills_wildshape_charges true
lp group class_druid permission set placeholderapi.placeholder.skills_druid_longest_cd true
