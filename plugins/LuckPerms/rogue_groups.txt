# LuckPerms - Classe Ladino e subclasses
lp creategroup class_rogue
lp group class_rogue setdisplayname "Ladino"
lp group class_rogue meta add prefix "&8[Ladino]&f "
lp group class_rogue permission set class_rogue true
lp group class_rogue permission set class.rogue true
lp group class_rogue permission set skillapi.class.rogue true
lp group class_rogue permission set mythicmobs.item.rogue true
lp group class_rogue permission set skript.command.furtivo true
lp group class_rogue permission set skript.command.evasao true
lp group class_rogue permission set skript.command.pique true
lp group class_rogue permission set skript.command.kit_ladino true
lp group class_rogue permission set skript.command.finese true
lp group class_rogue permission set group.class_rogue true

lp group class_rogue.thief permission set class_rogue.thief true

lp group class_rogue.thief permission set mythicmobs.item.rogue.thief true
