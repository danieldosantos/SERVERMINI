# Grupos LuckPerms para a classe Clérigo
lp creategroup class_cleric
lp creategroup class_cleric.life
lp creategroup class_cleric.war
lp creategroup class_cleric.twilight

# Herdar classe base para os domínios
lp group class_cleric.life parent add class_cleric
lp group class_cleric.war parent add class_cleric
lp group class_cleric.twilight parent add class_cleric

# Permissões gerais da classe
lp group class_cleric permission set skillapi.class.cleric true
lp group class_cleric permission set skillapi.skill.palavra_curativa true
lp group class_cleric permission set skillapi.skill.protecao_sagrada true
lp group class_cleric permission set skillapi.skill.aura_de_fe true
lp group class_cleric permission set skillapi.skill.canalizar_divindade true
