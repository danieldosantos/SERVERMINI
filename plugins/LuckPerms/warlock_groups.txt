# LuckPerms configuração de grupos para classe Bruxo
lp creategroup class_warlock
lp group class_warlock setdisplayname "Bruxo"
lp group class_warlock meta add prefix "&5[Bruxo]&f "
lp group class_warlock permission set class.warlock true
lp group class_warlock permission set skillapi.class.warlock true
lp group class_warlock permission set mythicmobs.item.warlock true
lp group class_warlock permission set skript.command.hex true
lp group class_warlock permission set skript.command.eldritch true
lp group class_warlock permission set skript.command.kit_bruxo true
lp group class_warlock permission set raceseffects.abilities true
lp group class_warlock permission set placeholderapi.placeholder.skills_warlock_hex_stacks true
