# LuckPerms setup for Bárbaro and subclasses
lp creategroup class_barbarian
lp group class_barbarian setdisplayname "Bárbaro"
lp group class_barbarian permission set skillapi.class.barbarian true
lp group class_barbarian permission set skillapi.access.barbarian true
lp group class_barbarian permission set group.class_barbarian true

lp creategroup class_barbarian.berserker
lp group class_barbarian.berserker setdisplayname "Bárbaro - Berserker"
lp group class_barbarian.berserker parent set class_barbarian
lp group class_barbarian.berserker permission set group.class_barbarian.berserker true
lp group class_barbarian.berserker permission set skillapi.subclass.barbarian.berserker true

lp creategroup class_barbarian.totem
lp group class_barbarian.totem setdisplayname "Bárbaro - Guardião Totêmico"
lp group class_barbarian.totem parent set class_barbarian
lp group class_barbarian.totem permission set group.class_barbarian.totem true
lp group class_barbarian.totem permission set skillapi.subclass.barbarian.totem true

lp creategroup class_barbarian.storm
lp group class_barbarian.storm setdisplayname "Bárbaro - Arauto da Tempestade"
lp group class_barbarian.storm parent set class_barbarian
lp group class_barbarian.storm permission set group.class_barbarian.storm true
lp group class_barbarian.storm permission set skillapi.subclass.barbarian.storm true
