# LuckPerms - configuração de grupos para classe Monge
lp creategroup class_monk
lp creategroup class_monk.open_hand
lp creategroup class_monk.shadow
lp creategroup class_monk.elements

# Heranças
lp group class_monk.open_hand parents add class_monk
lp group class_monk.shadow parents add class_monk
lp group class_monk.elements parents add class_monk
